library(pandoc)
# Install version
pandoc_install("2.7.3")

#>   Use 'force = TRUE' to overwrite.
pandoc_install("2.11.4")

#>   Use 'force = TRUE' to overwrite.
# Highest install is used
pandoc_version()

FILES=*.rst
for f in $FILES
do
  # extension="${f##*.}"
  filename="${f%.*}"
  echo "Converting $f to $filename.rst"
  `pandoc $f -t markdown -o $filename.md`

  # rm $f
done
